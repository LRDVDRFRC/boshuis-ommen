4 Backend implementatie

// api/create-payment.js (Vercel serverless function)
const { createMollieClient } = require('@mollie/api-client');

const mollieClient = createMollieClient({
  apiKey: process.env.MOLLIE_API_KEY  // bewaar in Vercel env vars!
});

module.exports = async (req, res) => {
  if (req.method !== 'POST') return res.status(405).end();

  const { checkin, checkout, guests, name, email, nights, totalPrice } = req.body;

  try {
    const payment = await mollieClient.payments.create({
      amount: {
        currency: 'EUR',
        value: totalPrice.toFixed(2)  // bijv. "285.00"
      },
      description: `Boshuis Ommen: ${checkin} t/m ${checkout} (${nights} nachten)`,
      redirectUrl: `${process.env.BASE_URL}/bevestiging?booking_id=${bookingId}`,
      webhookUrl: `${process.env.BASE_URL}/api/mollie-webhook`,
      metadata: {
        checkin, checkout, guests, name, email, nights
      }
    });

    // Sla boeking op met status 'pending'
    await saveBooking({
      id: payment.id,
      checkin, checkout, guests, name, email, nights,
      totalPrice, status: 'pending',
      createdAt: new Date().toISOString()
    });

    res.json({ checkoutUrl: payment.getCheckoutUrl() });
  } catch (error) {
    console.error('Mollie error:', error);
    res.status(500).json({ error: 'Betaling aanmaken mislukt' });
  }
};

5 Webhook (betaling bevestiging)

// api/mollie-webhook.js
const { createMollieClient } = require('@mollie/api-client');

const mollieClient = createMollieClient({
  apiKey: process.env.MOLLIE_API_KEY
});

module.exports = async (req, res) => {
  const { id } = req.body;  // Mollie stuurt payment ID

  const payment = await mollieClient.payments.get(id);

  if (payment.status === 'paid') {
    const { checkin, checkout, name, email, guests, nights } = payment.metadata;

    // 1. Update boeking status
    await updateBookingStatus(id, 'confirmed');

    // 2. Stuur bevestigingsmail
    await sendConfirmationEmail({ name, email, checkin, checkout, guests, nights,
      totalPrice: payment.amount.value });

    // 3. Blokkeer datums in kalender
    await addToCalendar({ checkin, checkout, name });
  }

  res.status(200).end();  // Mollie verwacht 200 OK
};

4. Kalender Synchronisatie (iCal)

Twee-weg synchronisatie voorkomt dubbele boekingen tussen je eigen site en Natuurhuisje.

Stap A: Importeer Natuurhuisje kalender

Natuurhuisje biedt een iCal-export URL aan voor elke advertentie. Ga naar:

Natuurhuisje Dashboard â Je advertentie (68313) â Kalender â iCal export

// api/sync-calendar.js â draait elk uur via cron
const ical = require('node-ical');

async function importNatuurhuisjeBookings() {
  // URL uit je Natuurhuisje dashboard
  const ICAL_URL = process.env.NATUURHUISJE_ICAL_URL;

  const events = await ical.async.fromURL(ICAL_URL);

  const bookedDates = [];

  for (const event of Object.values(events)) {
    if (event.type === 'VEVENT') {
      bookedDates.push({
        start: event.start.toISOString().split('T')[0],
        end: event.end.toISOString().split('T')[0],
        summary: event.summary || 'Natuurhuisje boeking',
        source: 'natuurhuisje'
      });
    }
  }

  // Sla op en combineer met eigen boekingen
  await updateBlockedDates(bookedDates);
  return bookedDates;
}

Stap B: Exporteer eigen kalender naar Natuurhuisje

// api/calendar.ics â publiek endpoint

module.exports = async (req, res) => {
  const bookings = await getConfirmedBookings();

  const cal = icalGenerator({
    name: 'Boshuis Ommen',
    prodId: '//Boshuis Ommen//Direct Bookings//NL'
  });

  for (const booking of bookings) {
    cal.createEvent({
      start: new Date(booking.checkin),
      end: new Date(booking.checkout),
      summary: `Geboekt: ${booking.name}`,
      description: `Direct boeking via boshuisommen.nl`
    });
  }

  res.setHeader('Content-Type', 'text/calendar');
  res.send(cal.toString());
};

// api/cron/pre-arrival.js â draait dagelijks om 09:00
module.exports = async (req, res) => {
  // Vind boekingen met check-in over 2-3 dagen
  const twoDaysFromNow = addDays(new Date(), 2);
  const threeDaysFromNow = addDays(new Date(), 3);

  const upcomingBookings = await getBookings({
    checkin: { between: [twoDaysFromNow, threeDaysFromNow] },
    preArrivalSent: false,
    status: 'confirmed'
  });

  for (const booking of upcomingBookings) {
    await sendPreArrivalEmail(booking);
    await markPreArrivalSent(booking.id);
  }

  res.json({ sent: upcomingBookings.length });
};

// api/cron/post-stay.js â draait dagelijks om 10:00
module.exports = async (req, res) => {
  const yesterday = addDays(new Date(), -1);

  const completedStays = await getBookings({
    checkout: yesterday.toISOString().split('T')[0],
    postStaySent: false,
    status: 'confirmed'
  });

  for (const booking of completedStays) {
    await sendPostStayEmail(booking);
    await markPostStaySent(booking.id);
  }

  res.json({ sent: completedStays.length });
};
